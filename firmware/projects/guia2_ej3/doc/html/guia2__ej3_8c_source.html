101void EnviarDistanciaUART(uint16_t distancia) {
102 char buffer[10];
103
104 if (distancia > 999) distancia = 999;
105
106 buffer[0] = (distancia / 100) % 10 + '0';
107 buffer[1] = (distancia / 10) % 10 + '0';
108 buffer[2] = (distancia % 10) + '0';
109 buffer[3] = ' ';

111 buffer[5] = 'm';
112 buffer[6] = '\r';
113 buffer[7] = '\n';
114 buffer[8] = '\0';

132static void TareaMedicion(void *params) {
133 uint8_t dato;
134
135 while (1) {
136 if (UartReadByte(UART_PC, &dato)) {
137 if (dato == 'O' || dato == 'o') {
138 medir = !medir;
139 } else if (dato == 'H' || dato == 'h') {
140 hold = !hold;
141 }
142 }
143
144 if (ulTaskNotifyTake(pdTRUE, 0)) {
145 if (medir) {
146 uint16_t distancia = HcSr04ReadDistanceInCentimeters();
147
148 EnviarDistanciaUART(distancia);
149
150 if (!hold) {
151 LcdItsE0803Write(distancia);
152 }
153
154 ActualizarLedsPorDistancia(distancia);
155
156 } else {
157 LedsOffAll();
158 LcdItsE0803Off();
159 }
160 }
161
162 vTaskDelay(pdMS_TO_TICKS(10));
163 }
164}
